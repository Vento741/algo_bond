Секция 4: Система агентов Claude Code

  .claude/
  ├── agents/                            # Специализированные агенты
  │   ├── orchestrator.md                # Оркестратор — управление всеми агентами
  │   ├── backend-dev.md                 # Backend-разработчик (FastAPI, DB, API)
  │   ├── frontend-dev.md                # Frontend-разработчик (React, Charts, UI)
  │   ├── researcher.md                  # Ресёрчер (поиск решений, библиотек)
  │   ├── trader.md                      # Трейдер-агент (Bybit, фьючерсы, стратегии)
  │   ├── algorithm-engineer.md          # Алгоритмист (Pine→Python, ML, индикаторы)
  │   ├── debugger.md                    # Дебаггер (диагностика, фикс ошибок)
  │   ├── code-reviewer.md              # Ревьюер/Рефакторер
  │   ├── consultant.md                  # Консультант (архитектура, бизнес-логика)
  │   └── market-analyst.md             # Аналитик (WebSocket, гипотезы, стратегии)
  │
  ├── skills/                            # Скиллы (вызываются через /команду)
  │   ├── deploy.md                      # /deploy — деплой на VPS
  │   ├── backtest.md                    # /backtest — запуск бэктеста
  │   ├── strategy-test.md               # /strategy-test — тест гипотезы
  │   ├── db-migrate.md                  # /db-migrate — миграции Alembic
  │   ├── market-check.md                # /market-check — проверка рыночных данных
  │   ├── bot-control.md                 # /bot-control — управление ботами
  │   ├── pine-convert.md                # /pine-convert — конвертация Pine→Python
  │   └── changelog.md                   # /changelog — обновление CHANGELOG.md
  │
  ├── hooks/
  │   ├── pre-commit.sh                  # Линтинг, форматирование, проверка секретов
  │   ├── post-push.sh                   # Уведомление о пуше, триггер деплоя
  │   └── on-error.sh                    # Логирование ошибок агентов
  │
  ├── commands/
  │   ├── status.md                      # /status — статус всех сервисов
  │   ├── logs.md                        # /logs — просмотр логов VPS
  │   └── health.md                      # /health — healthcheck платформы
  │
  ├── settings.json                      # Права, плагины, env
  └── CLAUDE.md                          # Главные инструкции проекта

  Описание каждого агента

  ┌────────────────────┬───────────────────────────────────────────────────────────────────┬───────────────────────────────────────────────────────────────────────────────┐
  │       Агент        │                               Роль                                │                              Ключевые инструкции                              │
  ├────────────────────┼───────────────────────────────────────────────────────────────────┼───────────────────────────────────────────────────────────────────────────────┤
  │ orchestrator       │ Координирует агентов, декомпозирует задачи, контролирует прогресс │ Знает всех агентов, умеет делегировать, следит за CHANGELOG                   │
  ├────────────────────┼───────────────────────────────────────────────────────────────────┼───────────────────────────────────────────────────────────────────────────────┤
  │ backend-dev        │ FastAPI, SQLAlchemy, Alembic, Celery tasks, API endpoints         │ Модульная архитектура, типизация, docstrings на RU                            │
  ├────────────────────┼───────────────────────────────────────────────────────────────────┼───────────────────────────────────────────────────────────────────────────────┤
  │ frontend-dev       │ React, TradingView Charts, Shadcn/UI, Zustand                     │ Все дизайн-правила (luxury fintech / trading terminal), Lucide Icons, адаптив │
  ├────────────────────┼───────────────────────────────────────────────────────────────────┼───────────────────────────────────────────────────────────────────────────────┤
  │ researcher         │ Поиск решений на GitHub, npm, PyPI, документации                  │ Глубокий анализ, сравнение альтернатив, адаптация под проект                  │
  ├────────────────────┼───────────────────────────────────────────────────────────────────┼───────────────────────────────────────────────────────────────────────────────┤
  │ trader             │ Bybit API, фьючерсы, ордера, позиции, risk management             │ Знание pybit, USDT-M фьючерсов, ликвидаций, маржи, funding rate               │
  ├────────────────────┼───────────────────────────────────────────────────────────────────┼───────────────────────────────────────────────────────────────────────────────┤
  │ algorithm-engineer │ Pine Script → Python, ML-модели, индикаторы, KNN                  │ Портирование стратегий, оптимизация, numpy/pandas                             │
  ├────────────────────┼───────────────────────────────────────────────────────────────────┼───────────────────────────────────────────────────────────────────────────────┤
  │ debugger           │ Диагностика ошибок, трейсы, профилирование                        │ Анализ логов, docker logs, Redis/PG диагностика                               │
  ├────────────────────┼───────────────────────────────────────────────────────────────────┼───────────────────────────────────────────────────────────────────────────────┤
  │ code-reviewer      │ Code review, рефакторинг, качество кода                           │ PEP8, SOLID, security, performance, type hints                                │
  ├────────────────────┼───────────────────────────────────────────────────────────────────┼───────────────────────────────────────────────────────────────────────────────┤
  │ consultant         │ Архитектурные решения, бизнес-логика, масштабирование             │ Знание домена алготрейдинга, SaaS-паттерны                                    │
  ├────────────────────┼───────────────────────────────────────────────────────────────────┼───────────────────────────────────────────────────────────────────────────────┤
  │ market-analyst     │ WebSocket подписки, проверка гипотез, анализ рынка                │ Bybit WS API, свечные паттерны, бэктест гипотез                               │
  └────────────────────┴───────────────────────────────────────────────────────────────────┴───────────────────────────────────────────────────────────────────────────────┘

  Описание скиллов

  ┌───────────────┬──────────────────────────────────┬────────────────────────────────────────────────────────────┐
  │     Скилл     │             Команда              │                         Что делает                         │
  ├───────────────┼──────────────────────────────────┼────────────────────────────────────────────────────────────┤
  │ deploy        │ /deploy [env]                    │ SSH на VPS, git pull, docker-compose build/up, healthcheck │
  ├───────────────┼──────────────────────────────────┼────────────────────────────────────────────────────────────┤
  │ backtest      │ /backtest SYMBOL TF START END    │ Запуск бэктеста стратегии через API или локально           │
  ├───────────────┼──────────────────────────────────┼────────────────────────────────────────────────────────────┤
  │ strategy-test │ /strategy-test                   │ Подписка на WS, прогон стратегии на live-данных, отчёт     │
  ├───────────────┼──────────────────────────────────┼────────────────────────────────────────────────────────────┤
  │ db-migrate    │ /db-migrate [message]            │ Alembic revision + upgrade                                 │
  ├───────────────┼──────────────────────────────────┼────────────────────────────────────────────────────────────┤
  │ market-check  │ /market-check SYMBOL             │ Текущая цена, объём, тренд, сигналы стратегии              │
  ├───────────────┼──────────────────────────────────┼────────────────────────────────────────────────────────────┤
  │ bot-control   │ /bot-control [start|stop|status] │ Управление торговыми ботами на VPS                         │
  ├───────────────┼──────────────────────────────────┼────────────────────────────────────────────────────────────┤
  │ pine-convert  │ /pine-convert [file]             │ Конвертация Pine Script → Python класс стратегии           │
  ├───────────────┼──────────────────────────────────┼────────────────────────────────────────────────────────────┤
  │ changelog     │ /changelog                       │ Автогенерация записи в CHANGELOG.md по последним коммитам  │
  └───────────────┴──────────────────────────────────┴────────────────────────────────────────────────────────────┘
